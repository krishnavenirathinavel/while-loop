 
#include <stdio.h>
int main()
{ 
    int n, i = 0, data;
    int gb;
    int ExhaustedDay = 0;
    int overused = 0;

    scanf("%d",&data);
    scanf("%d",&n);
    int remaining = data;
    while(i < n){
      scanf("%d",&gb);
      remaining = remaining - gb;
      if(remaining <= 0){
        ExhaustedDay = i + 1;
        if(remaining < 0){
          overused = -remaining;
          }
          break;
          }
          i++;
          }
          if(ExhaustedDay == 0){
            printf("Exhausted Day: Not Exhausted\n");
            printf("Overused Data: 0\n");
    } else {
        printf("Exhausted Day: %d\n",ExhaustedDay);
        printf("Overused Data: %d\n",overused);
    }

    return 0;
}
