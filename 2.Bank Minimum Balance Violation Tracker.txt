#include <stdio.h>
  int main() {
    int N, balance, t;
    int low = 0, i = 0;

    scanf("%d", &N);
    scanf("%d", &balance);
    while (i < N) {
      scanf("%d", &t);
      balance = balance + t;
      if (balance < 2000)
      low++;
      i++;
    }
    printf("Final Balance: %d\n", balance);
    printf("Low Balance Days: %d", low);

    return 0;
}
