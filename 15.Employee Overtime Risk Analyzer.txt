 
#include <stdio.h>
int main()
{
  int i = 0,n,days = 0;
  int total = 0;
  int burnout = 0;
  scanf("%d",&n);
  
  while(i < n){
     scanf("%d",&days);
       total =  total + days;
    if(days > 4){
     
      burnout++;
    }
    i++;
  }
  printf("Total Overtime:%d\n",total);
  printf("Burnout Days: %d\n",burnout);

    return 0;
}
