 #include <stdio.h>

int main() {
  int n, i = 1;
  scanf("%d", &n);
  int index[n];
  int totalDrops = 0;
  int consecutive = 0;
  int crashDay = -1;
  int j = 0;
  while(j < n){
    scanf("%d", &index[j]);
    j++;
    }
    while(i < n){
      if(index[i] < index[i-1]){
        totalDrops++;
        consecutive++;
        if(consecutive == 3 && crashDay == -1){
          crashDay = i + 1;
          }
        } else {
          consecutive = 0;  }
          i++;
    }
    if(crashDay == -1)
    printf("Crash Day: Not Detected\n");
    else
    printf("Crash Day: %d\n", crashDay);
    printf("Total Drops: %d\n", totalDrops);

    return 0;
}
