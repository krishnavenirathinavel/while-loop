#include <stdio.h>
int main()
{ 
   int n,i=0,days;
   int delayedcounts = 0;
   int totaldays = 0;
   scanf("%d",&n);
   while (i < n){
    scanf("%d",&days);
    totaldays = totaldays + days;
    if (days  >= 2){
    delayedcounts++;}
    i++;
   }
   printf("Total Delay : %d\n",totaldays);
   printf("Delayed Days: %d\n",delayedcounts);
    return 0;
  
}
